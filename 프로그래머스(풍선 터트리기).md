function solution(a) {
    let answer = 0
    let left = new Array(a.length)
    let right = new Array(a.length)
    
    let minimum = a[0]
    for(let i=0;i<a.length;i++){
        if(minimum > a[i]) minimum = a[i]
        left[i] = minimum
    }
    
    minimum = a[a.length -1]
    for(let i=a.length-1;i>=0;i--){
        if(minimum > a[i]) minimum = a[i]
        right[i] = minimum
    }
    
    for(let i =0;i<a.length;i++){
        if(a[i] <= left[i] || a[i] <= right[i]) answer++
    }
    return answer
}
